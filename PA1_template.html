df2$wday<-wday(df2[,2])
convert<-function(num){
  if(num>=2 & num<=6){return("weekday")}
  else if(num==1||num==7){return("weekend")}
  else{stop("input is out of range")}
}
df2$wdaychr<-sapply(df2$wday, convert)
